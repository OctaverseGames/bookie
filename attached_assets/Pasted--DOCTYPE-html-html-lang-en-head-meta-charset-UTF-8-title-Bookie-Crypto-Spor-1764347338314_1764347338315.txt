      const pill = document.createElement("button");

        const selectionId = `${sportKey}-${gameIndex}-${marketType}`;

        const existingIndex = selections.findIndex((s) => s.id === selectionId);
        if (existingIndex > -1) {
          selections.splice(existingIndex, 1);
        } else {
          const numericOdds =
            typeof value === "number" ? value : guessDecimalOdds(value);
          selections.push({
            id: selectionId,
            sport: sportKey,
            league: game.league,
            home: game.home,
            away: game.away,
            label,
            value,
            odds: numericOdds,
          });
        }

        renderBetslip();
      });

      return pill;
    }

        item.appendChild(meta);
        betslipBody.appendChild(item);
